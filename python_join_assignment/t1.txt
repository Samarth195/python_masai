-- create users table
CREATE TABLE users (
    user_id INT PRIMARY KEY,
    name VARCHAR(100),
    signup_date DATE
);

-- create orders table
CREATE TABLE orders (
    order_id INT PRIMARY KEY,
    user_id INT,
    amount INT
);

-- create activity table
CREATE TABLE activity (
    user_id INT,
    login_count INT
);

-- 1. user name and order amount using inner join
SELECT users.name, orders.amount
FROM users
INNER JOIN orders ON users.user_id = orders.user_id;

-- 2. user name and total order amount
SELECT users.name, SUM(orders.amount) AS total_order_amount
FROM users
INNER JOIN orders ON users.user_id = orders.user_id
GROUP BY users.name;

-- 3. all users and their login counts including users with no activity
SELECT users.name, activity.login_count
FROM users
LEFT JOIN activity ON users.user_id = activity.user_id;

-- 4. users who never placed an order
SELECT users.name
FROM users
LEFT JOIN orders ON users.user_id = orders.user_id
WHERE orders.order_id IS NULL;

-- 5. combining all 3 tables
SELECT users.user_id, users.name, SUM(orders.amount) AS total_order_amount, activity.login_count
FROM users
LEFT JOIN orders ON users.user_id = orders.user_id
LEFT JOIN activity ON users.user_id = activity.user_id
GROUP BY users.user_id, users.name, activity.login_count;

-- 6. users who placed orders greater than 400
SELECT users.name, orders.amount
FROM users
INNER JOIN orders ON users.user_id = orders.user_id
WHERE orders.amount > 400;

-- 7. top 3 users by total order amount
SELECT users.name, SUM(orders.amount) AS total_order_amount
FROM users
INNER JOIN orders ON users.user_id = orders.user_id
GROUP BY users.name
ORDER BY total_order_amount DESC
LIMIT 3;

-- 8. users sorted by total order amount and login count
SELECT users.name, SUM(orders.amount) AS total_order_amount, activity.login_count
FROM users
LEFT JOIN orders ON users.user_id = orders.user_id
LEFT JOIN activity ON users.user_id = activity.user_id
GROUP BY users.name, activity.login_count
ORDER BY total_order_amount DESC, activity.login_count DESC;

-- 9. average order amount per user
SELECT users.name, AVG(orders.amount) AS avg_order_amount
FROM users
INNER JOIN orders ON users.user_id = orders.user_id
GROUP BY users.name;

-- 10. combined query
SELECT users.user_id, users.name, SUM(orders.amount) AS total_order_amount, activity.login_count
FROM users
LEFT JOIN orders ON users.user_id = orders.user_id
LEFT JOIN activity ON users.user_id = activity.user_id
GROUP BY users.user_id, users.name, activity.login_count
ORDER BY total_order_amount DESC
LIMIT 3;
